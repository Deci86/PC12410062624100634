CREATE DATABASE TallerMecanico;
GO

USE TallerMecanico;
GO

-- 2. Creación de la tabla: TipoServicio
CREATE TABLE TipoServicio (
    ID INT IDENTITY(1,1) NOT NULL,
    nombre VARCHAR(100) NOT NULL,
    precioBase DECIMAL(10, 2) NOT NULL,
    CONSTRAINT PK_TipoServicio PRIMARY KEY (ID)
);
GO

-- 3. Creación de la tabla: Cliente
CREATE TABLE Cliente (
    ID INT IDENTITY(1,1) NOT NULL,
    paterno VARCHAR(50) NOT NULL,
    materno VARCHAR(50) NULL, -- Se permite NULL por si algún cliente no lo tiene
    nombres VARCHAR(100) NOT NULL,
    correo VARCHAR(150) NULL,
    telefono VARCHAR(20) NULL,
    CONSTRAINT PK_Cliente PRIMARY KEY (ID)
);
GO

-- 4. Creación de la tabla: Vehiculo (Necesaria para relacionar con OrdenServicio)
CREATE TABLE Vehiculo (
    ID INT IDENTITY(1,1) NOT NULL,
    Placa VARCHAR(15) NOT NULL,
    Marca VARCHAR(50) NOT NULL,
    Modelo VARCHAR(50) NOT NULL,
    ClienteId INT NOT NULL,
    CONSTRAINT PK_Vehiculo PRIMARY KEY (ID),
    CONSTRAINT FK_Vehiculo_Cliente FOREIGN KEY (ClienteId) REFERENCES Cliente(ID),
    CONSTRAINT UQ_Placa UNIQUE (Placa) -- Evita placas duplicadas
);
GO

-- 5. Creación de la tabla: OrdenServicio
CREATE TABLE OrdenServicio (
    ID INT IDENTITY(1,1) NOT NULL,
    FechaIngreso DATETIME NOT NULL DEFAULT GETDATE(), -- Por defecto toma la fecha actual
    DescripcionProblema VARCHAR(MAX) NOT NULL,
    CostoEstimado DECIMAL(10, 2) NOT NULL,
    Estado VARCHAR(30) NOT NULL DEFAULT 'Ingresado', -- Estado inicial por defecto
    VehiculoId INT NOT NULL,
    TipoServicioId INT NOT NULL,
    CONSTRAINT PK_OrdenServicio PRIMARY KEY (ID),
    CONSTRAINT FK_OrdenServicio_Vehiculo FOREIGN KEY (VehiculoId) REFERENCES Vehiculo(ID),
    CONSTRAINT FK_OrdenServicio_TipoServicio FOREIGN KEY (TipoServicioId) REFERENCES TipoServicio(ID)
);
GO
